17/// Wraps an object to be inserted in a linked list.
18pub trait Wrapper<T: ?Sized> {
19    /// Converts the wrapped object into a pointer that represents it.
20    fn into_pointer(self) -> NonNull<T>;
21
22    /// Converts the object back from the pointer representation.
23    ///
24    /// # Safety
25    ///
26    /// The passed pointer must come from a previous call to [`Wrapper::into_pointer()`].
27    unsafe fn from_pointer(ptr: NonNull<T>) -> Self;
28
29    /// Returns a reference to the wrapped object.
30    fn as_ref(&self) -> &T;
31}
32
33impl<T: ?Sized> Wrapper<T> for Box<T> {
34    #[inline]
35    fn into_pointer(self) -> NonNull<T> {
36        NonNull::new(Box::into_raw(self)).unwrap()
37    }
38
39    #[inline]
40    unsafe fn from_pointer(ptr: NonNull<T>) -> Self {
41        unsafe { Box::from_raw(ptr.as_ptr()) }
42    }
43
44    #[inline]
45    fn as_ref(&self) -> &T {
46        AsRef::as_ref(self)
47    }
48}
49
50impl<T: ?Sized> Wrapper<T> for Arc<T> {
51    #[inline]
52    fn into_pointer(self) -> NonNull<T> {
53        NonNull::new(Arc::into_raw(self) as _).unwrap()
54    }
55
56    #[inline]
57    unsafe fn from_pointer(ptr: NonNull<T>) -> Self {
58        // SAFETY: The safety requirements of `from_pointer` satisfy the ones from `Arc::from_raw`.
59        unsafe { Arc::from_raw(ptr.as_ptr() as _) }
60    }
61
62    #[inline]
63    fn as_ref(&self) -> &T {
64        AsRef::as_ref(self)
65    }
66}
67
68impl<T: ?Sized> Wrapper<T> for &T {
69    #[inline]
70    fn into_pointer(self) -> NonNull<T> {
71        NonNull::from(self)
72    }
73
74    #[inline]
75    unsafe fn from_pointer(ptr: NonNull<T>) -> Self {
76        unsafe { &*ptr.as_ptr() }
77    }
78
79    #[inline]
80    fn as_ref(&self) -> &T {
81        self
82    }
83}

85/// A descriptor of wrapped list elements.
86pub trait GetLinksWrapped: GetLinks {
87    /// Specifies which wrapper (e.g., `Box` and `Arc`) wraps the list entries.
88    type Wrapped: Wrapper<Self::EntryType>;
89}
90
91impl<T: ?Sized> GetLinksWrapped for Box<T>
92where
93    Box<T>: GetLinks,
94{
95    type Wrapped = Box<<Box<T> as GetLinks>::EntryType>;
96}
97
98impl<T: GetLinks + ?Sized> GetLinks for Box<T> {
99    type EntryType = T::EntryType;
100
101    #[inline]
102    fn get_links(data: &Self::EntryType) -> &Links<Self::EntryType> {
103        <T as GetLinks>::get_links(data)
104    }
105}
106
107impl<T: ?Sized> GetLinksWrapped for Arc<T>
108where
109    Arc<T>: GetLinks,
110{
111    type Wrapped = Arc<<Arc<T> as GetLinks>::EntryType>;
112}
113
114impl<T: GetLinks + ?Sized> GetLinks for Arc<T> {
115    type EntryType = T::EntryType;
116
117    #[inline]
118    fn get_links(data: &Self::EntryType) -> &Links<Self::EntryType> {
119        <T as GetLinks>::get_links(data)
120    }
121}

123/// A linked list.
124///
125/// Elements in the list are wrapped and ownership is transferred to the list while the element is
126/// in the list.
127pub struct List<G: GetLinksWrapped> {
128    list: RawList<G>,
129}
130
131impl<G: GetLinksWrapped> List<G> {
132    /// Constructs a new empty linked list.
133    pub const fn new() -> Self {
134        Self {
135            list: RawList::new(),
136        }
137    }
138
139    /// Returns an iterator for the list starting at the first entry.
140    pub fn iter(&self) -> Iterator<'_, G> {
141        Iterator::new(self)
142    }
143
144    /// Returns whether the list is empty.
145    pub const fn is_empty(&self) -> bool {
146        self.list.is_empty()
147    }
148
149    /// Adds the given object to the end (back) of the list.
150    ///
151    /// It is dropped if it's already on this (or another) list; this can happen for
152    /// reference-counted objects, so dropping means decrementing the reference count.
153    pub fn push_back(&mut self, data: G::Wrapped) {
154        let ptr = data.into_pointer();
155
156        // SAFETY: We took ownership of the entry, so it is safe to insert it.
157        if !unsafe { self.list.push_back(ptr) } {
158            // If insertion failed, rebuild object so that it can be freed.
159            // SAFETY: We just called `into_pointer` above.
160            unsafe { G::Wrapped::from_pointer(ptr) };
161        }
162    }
163
164    /// Adds the given object to the first (front) of the list.
165    ///
166    /// It is dropped if it's already on this (or another) list; this can happen for
167    /// reference-counted objects, so dropping means decrementing the reference count.
168    pub fn push_front(&mut self, data: G::Wrapped) {
169        let ptr = data.into_pointer();
170
171        // SAFETY: We took ownership of the entry, so it is safe to insert it.
172        if !unsafe { self.list.push_front(ptr) } {
173            // If insertion failed, rebuild object so that it can be freed.
174            unsafe { G::Wrapped::from_pointer(ptr) };
175        }
176    }
177
178    /// Inserts the given object after `existing`. Returns true if it's successfully inserted.
179    ///
180    /// It is dropped if it's already on this (or another) list; this can happen for
181    /// reference-counted objects, so dropping means decrementing the reference count.
182    ///
183    /// # Safety
184    ///
185    /// Callers must ensure that `existing` points to a valid entry that is on the list.
186    pub unsafe fn insert_after(
187        &mut self,
188        existing: NonNull<G::EntryType>,
189        data: G::Wrapped,
190    ) -> bool {
191        let ptr = data.into_pointer();
192        let inserted = unsafe { self.list.insert_after(existing, ptr) };
193        if !inserted {
194            // If insertion failed, rebuild object so that it can be freed.
195            unsafe { G::Wrapped::from_pointer(ptr) };
196        }
197        inserted
198    }
199
200    /// Removes the given entry.
201    ///
202    /// # Safety
203    ///
204    /// Callers must ensure that `data` is either on this list or in no list. It being on another
205    /// list leads to memory unsafety.
206    pub unsafe fn remove(&mut self, data: &G::Wrapped) -> Option<G::Wrapped> {
207        let entry_ref = Wrapper::as_ref(data);
208        if unsafe { self.list.remove(entry_ref) } {
209            Some(unsafe { G::Wrapped::from_pointer(NonNull::from(entry_ref)) })
210        } else {
211            None
212        }
213    }
214
215    /// Removes the element currently at the front of the list and returns it.
216    ///
217    /// Returns `None` if the list is empty.
218    pub fn pop_front(&mut self) -> Option<G::Wrapped> {
219        let front = self.list.pop_front()?;
220        // SAFETY: Elements on the list were inserted after a call to `into_pointer `.
221        Some(unsafe { G::Wrapped::from_pointer(front) })
222    }
223
224    /// Returns an immutable cursor starting on the first (front) element of the list.
225    pub fn cursor_front(&self) -> Cursor<'_, G> {
226        Cursor::new(self.list.cursor_front())
227    }
228
229    /// Returns a mutable cursor starting on the first (front) element of the list.
230    pub fn cursor_front_mut(&mut self) -> CursorMut<'_, G> {
231        CursorMut::new(self.list.cursor_front_mut())
232    }
233}
234
235impl<G: GetLinksWrapped> Default for List<G> {
236    fn default() -> Self {
237        Self::new()
238    }
239}
240
241impl<G: GetLinksWrapped> Drop for List<G> {
242    fn drop(&mut self) {
243        while self.pop_front().is_some() {}
244    }
245}

247/// A list cursor that allows traversing a linked list and inspecting elements.
248pub struct Cursor<'a, G: GetLinksWrapped> {
249    cursor: raw_list::Cursor<'a, G>,
250}
251
252impl<'a, G: GetLinksWrapped> Cursor<'a, G> {
253    const fn new(cursor: raw_list::Cursor<'a, G>) -> Self {
254        Self { cursor }
255    }
256
257    /// Returns the element the cursor is currently positioned on.
258    pub fn current(&self) -> Option<&G::EntryType> {
259        self.cursor.current()
260    }
261
262    /// Returns the element pointer the cursor is currently positioned on.
263    pub fn current_ptr(&self) -> Option<NonNull<G::EntryType>> {
264        self.cursor.current_ptr()
265    }
266
267    /// Returns the element immediately after the one the cursor is positioned on.
268    pub fn peek_next(&self) -> Option<&G::EntryType> {
269        self.cursor.peek_next()
270    }
271
272    /// Returns the element immediately before the one the cursor is positioned on.
273    pub fn peek_prev(&self) -> Option<&G::EntryType> {
274        self.cursor.peek_prev()
275    }
276
277    /// Moves the cursor to the next element.
278    pub fn move_next(&mut self) {
279        self.cursor.move_next();
280    }
281}

283/// A list cursor that allows traversing a linked list and inspecting & mutating elements.
284pub struct CursorMut<'a, G: GetLinksWrapped> {
285    cursor: raw_list::CursorMut<'a, G>,
286}
287
288impl<'a, G: GetLinksWrapped> CursorMut<'a, G> {
289    const fn new(cursor: raw_list::CursorMut<'a, G>) -> Self {
290        Self { cursor }
291    }
292
293    /// Returns the element the cursor is currently positioned on.
294    ///
295    /// # Safety
296    ///
297    /// For `Box` or `Arc` that has unique access to the data, the method is safe.
298    /// For `Arc` whose strong count is not 1, the method is not safe because it
299    /// violates the safety requirements of [`Arc`].
300    pub unsafe fn current_mut(&mut self) -> Option<&mut G::EntryType> {
301        self.cursor.current_mut()
302    }
303
304    /// Returns the element the cursor is currently positioned on.
305    pub fn current(&self) -> Option<&G::EntryType> {
306        self.cursor.current()
307    }
308
309    /// Returns the element pointer the cursor is currently positioned on.
310    pub fn current_ptr(&self) -> Option<NonNull<G::EntryType>> {
311        self.cursor.current_ptr()
312    }
313
314    /// Insert data after the current node.
315    ///
316    /// If there is no current node, the data will be dropped without insertion.
317    pub fn insert_after(&mut self, data: G::Wrapped) -> bool {
318        if let Some(cur) = self.current_ptr() {
319            let new = data.into_pointer();
320            // SAFETY: the current pointer is valid from the list.
321            return unsafe { self.cursor.list.insert_after(cur, new) };
322        }
323        false
324    }
325
326    /// Removes the element the cursor is currently positioned on.
327    ///
328    /// After removal, it advances the cursor to the next element.
329    pub fn remove_current(&mut self) -> Option<G::Wrapped> {
330        let ptr = self.cursor.remove_current()?;
331
332        // SAFETY: Elements on the list were inserted after a call to `into_pointer `.
333        Some(unsafe { G::Wrapped::from_pointer(ptr) })
334    }
335
336    /// Returns the element immediately after the one the cursor is positioned on.
337    ///
338    /// # Safety
339    ///
340    /// For `Box` or `Arc` that has unique access to the data behind, the method is safe.
341    /// For `Arc` whose strong count is not 1, the method is not safe because it
342    /// violates the safety requirements of [`Arc`].
343    pub unsafe fn peek_next(&mut self) -> Option<&mut G::EntryType> {
344        self.cursor.peek_next()
345    }
346
347    /// Returns the element immediately before the one the cursor is positioned on.
348    ///
349    /// # Safety
350    ///
351    /// For `Box` or `Arc` that has unique access to the data behind, the method is safe.
352    /// For `Arc` whose strong count is not 1, the method is not safe because it
353    /// violates the safety requirements of [`Arc`].
354    pub unsafe fn peek_prev(&mut self) -> Option<&mut G::EntryType> {
355        self.cursor.peek_prev()
356    }
357
358    /// Moves the cursor to the next element.
359    pub fn move_next(&mut self) {
360        self.cursor.move_next();
361    }
362}

364/// An iterator for the linked list.
365pub struct Iterator<'a, G: GetLinksWrapped> {
366    iter: raw_list::Iterator<'a, G>,
367}
368
369impl<'a, G: GetLinksWrapped> Iterator<'a, G> {
370    fn new(list: &'a List<G>) -> Self {
371        Self {
372            iter: list.list.iter(),
373        }
374    }
375}
376
377impl<'a, G: GetLinksWrapped> iter::Iterator for Iterator<'a, G> {
378    type Item = &'a G::EntryType;
379
380    fn next(&mut self) -> Option<Self::Item> {
381        self.iter.next()
382    }
383}
384
385impl<G: GetLinksWrapped> iter::DoubleEndedIterator for Iterator<'_, G> {
386    fn next_back(&mut self) -> Option<Self::Item> {
387        self.iter.next_back()
388    }
389}

391#[cfg(test)]
392mod tests {
393    use super::{GetLinks, Links, List};
394
395    struct Example {
396        inner: usize,
397        links: Links<Self>,
398    }
399
400    impl GetLinks for Example {
401        type EntryType = Self;
402        fn get_links(obj: &Self) -> &Links<Self> {
403            &obj.links
404        }
405    }
406
407    #[track_caller]
408    fn assert_list_contents(list: &List<Box<Example>>, n: usize) {
409        // Assert that the list is ok going forward.
410        let mut count = 0;
411        for (i, e) in list.iter().enumerate() {
412            assert_eq!(i + 1, e.inner);
413            count += 1;
414        }
415        assert_eq!(count, n);
416
417        // Assert that the list is ok going backwards.
418        let mut count = n;
419        for e in list.iter().rev() {
420            assert_eq!(count, e.inner);
421            count -= 1;
422        }
423        assert_eq!(count, 0);
424    }
425
426    #[track_caller]
427    #[test]
428    fn test_push_back() {
429        const MAX: usize = 10;
430        let mut list = List::<Box<Example>>::new();
431
432        for n in 1..=MAX {
433            list.push_back(Box::new(Example {
434                inner: n,
435                links: Links::new(),
436            }));
437        }
438        assert_list_contents(&list, MAX);
439    }
