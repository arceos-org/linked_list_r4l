16/// A descriptor of list elements.
17///
18/// It describes the type of list elements and provides a function to determine how to get the
19/// links to be used on a list.
20///
21/// A type that may be in multiple lists simultaneously needs to implement one of these for each
22/// simultaneous list.
23pub trait GetLinks {
24    /// The type of the entries in the list.
25    type EntryType: ?Sized;
26
27    /// Returns the links to be used when linking an entry within a list.
28    fn get_links(data: &Self::EntryType) -> &Links<Self::EntryType>;
29}
30
31/// The links used to link an object on a linked list.
32///
33/// Instances of this type are usually embedded in structures and returned in calls to
34/// [`GetLinks::get_links`].
35pub struct Links<T: ?Sized> {
36    inserted: AtomicBool,
37    entry: UnsafeCell<ListEntry<T>>,
38}
39
40// SAFETY: `Links` can be safely sent to other threads but we restrict it to being `Send` only when
41// the list entries it points to are also `Send`.
42unsafe impl<T: ?Sized> Send for Links<T> {}
43
44// SAFETY: `Links` is usable from other threads via references but we restrict it to being `Sync`
45// only when the list entries it points to are also `Sync`.
46unsafe impl<T: ?Sized> Sync for Links<T> {}
47
48impl<T: ?Sized> Links<T> {
49    /// Constructs a new [`Links`] instance that isn't inserted on any lists yet.
50    pub const fn new() -> Self {
51        Self {
52            inserted: AtomicBool::new(false),
53            entry: UnsafeCell::new(ListEntry::new()),
54        }
55    }
56
57    fn acquire_for_insertion(&self) -> bool {
58        self.inserted
59            .compare_exchange(false, true, Ordering::Acquire, Ordering::Relaxed)
60            .is_ok()
61    }
62
63    fn release_after_removal(&self) {
64        self.inserted.store(false, Ordering::Release);
65    }
66}
67
68impl<T: ?Sized> Default for Links<T> {
69    fn default() -> Self {
70        Self::new()
71    }
72}

74struct ListEntry<T: ?Sized> {
75    next: Option<NonNull<T>>,
76    prev: Option<NonNull<T>>,
77}
78
79impl<T: ?Sized> ListEntry<T> {
80    const fn new() -> Self {
81        Self {
82            next: None,
83            prev: None,
84        }
85    }
86}

88/// A linked list.
89///
90/// # Invariants
91///
92/// The links of objects added to a list are owned by the list.
93pub struct RawList<G: GetLinks> {
94    head: Option<NonNull<G::EntryType>>,
95}
96
97impl<G: GetLinks> RawList<G> {
98    /// Constructs a new empty RawList.
99    pub const fn new() -> Self {
100        Self { head: None }
101    }
102
103    /// Returns an iterator for the list starting at the first entry.
104    pub fn iter(&self) -> Iterator<'_, G> {
105        Iterator::new(self.cursor_front(), self.cursor_back())
106    }
107
108    /// Returns whether the RawList is empty.
109    pub const fn is_empty(&self) -> bool {
110        self.head.is_none()
111    }
112
113    fn insert_after_priv(
114        &mut self,
115        existing: NonNull<G::EntryType>,
116        new_entry: &mut ListEntry<G::EntryType>,
117        new_ptr: Option<NonNull<G::EntryType>>,
118    ) {
119        {
120            // SAFETY: It's safe to get the previous entry of `existing` because the list cannot
121            // change.
122            let existing_links = unsafe { &mut *G::get_links(existing.as_ref()).entry.get() };
123            new_entry.next = existing_links.next;
124            existing_links.next = new_ptr;
125        }
126
127        new_entry.prev = Some(existing);
128
129        // SAFETY: It's safe to get the next entry of `existing` because the list cannot change.
130        let next_links =
131            unsafe { &mut *G::get_links(new_entry.next.unwrap().as_ref()).entry.get() };
132        next_links.prev = new_ptr;
133    }
134
135    /// Inserts the given object after `existing`.
136    ///
137    /// # Safety
138    ///
139    /// Callers must ensure that `existing` points to a valid entry that is on the list.
140    pub unsafe fn insert_after(
141        &mut self,
142        existing: NonNull<G::EntryType>,
143        new: NonNull<G::EntryType>,
144    ) -> bool {
145        // SAFETY: the caller ensures new is valid.
146        let links = unsafe { G::get_links(new.as_ref()) };
147        if !links.acquire_for_insertion() {
148            // Nothing to do if already inserted.
149            return false;
150        }
151
152        // SAFETY: The links are now owned by the list, so it is safe to get a mutable reference.
153        let new_entry = unsafe { &mut *links.entry.get() };
154        self.insert_after_priv(existing, new_entry, Some(new));
155        true
156    }
157
158    fn push_back_internal(&mut self, new: NonNull<G::EntryType>, front: bool) -> bool {
159        // SAFETY: We just get the valid EntryType ptr from the caller.
160        let links = unsafe { G::get_links(new.as_ref()) };
161        if !links.acquire_for_insertion() {
162            // Nothing to do if already inserted.
163            return false;
164        }
165
166        // SAFETY: The links are now owned by the list, so it is safe to get a mutable reference.
167        let new_entry = unsafe { &mut *links.entry.get() };
168        let new_ptr = Some(new);
169        match self.back() {
170            // SAFETY: `back` is valid as the list cannot change.
171            Some(back) => {
172                self.insert_after_priv(back, new_entry, new_ptr);
173                // if push front, update head
174                if front {
175                    self.head = new_ptr;
176                }
177            }
178            None => {
179                self.head = new_ptr;
180                new_entry.next = new_ptr;
181                new_entry.prev = new_ptr;
182            }
183        }
184        true
185    }
186
187    /// Adds the given object to the end (back) of the list.
188    ///
189    /// Rawlist will save the reference as node ptr.
190    /// The caller must ensure the validity of the reference while it is on
191    /// the linked list.
192    pub unsafe fn push_back(&mut self, new: NonNull<G::EntryType>) -> bool {
193        self.push_back_internal(new, false)
194    }
195
196    /// Adds the given object to the first (front) of the list.
197    ///
198    /// Rawlist will save the reference as node ptr.
199    /// The caller must ensure the validity of the reference while it is on
200    /// the linked list.
201    pub unsafe fn push_front(&mut self, new: NonNull<G::EntryType>) -> bool {
202        self.push_back_internal(new, true)
203    }
204
205    fn remove_internal(&mut self, data: &G::EntryType) -> bool {
206        let links = G::get_links(data);
207
208        // SAFETY: The links are now owned by the list, so it is safe to get a mutable reference.
209        let entry = unsafe { &mut *links.entry.get() };
210        let next = if let Some(next) = entry.next {
211            next
212        } else {
213            // Nothing to do if the entry is not on the list.
214            return false;
215        };
216
217        if ptr::eq(data, next.as_ptr()) {
218            // We're removing the only element.
219            self.head = None
220        } else {
221            // Update the head if we're removing it.
222            if let Some(raw_head) = self.head {
223                if ptr::eq(data, raw_head.as_ptr()) {
224                    self.head = Some(next);
225                }
226            }
227
228            // SAFETY: It's safe to get the previous entry because the list cannot change.
229            unsafe { &mut *G::get_links(entry.prev.unwrap().as_ref()).entry.get() }.next =
230                entry.next;
231
232            // SAFETY: It's safe to get the next entry because the list cannot change.
233            unsafe { &mut *G::get_links(next.as_ref()).entry.get() }.prev = entry.prev;
234        }
235
236        // Reset the links of the element we're removing so that we know it's not on any list.
237        entry.next = None;
238        entry.prev = None;
239        links.release_after_removal();
240        true
241    }
242
243    /// Removes the given entry.
244    ///
245    /// # Safety
246    ///
247    /// Callers must ensure that `data` is either on this list or in no list. It being on another
248    /// list leads to memory unsafety.
249    pub unsafe fn remove(&mut self, data: &G::EntryType) -> bool {
250        self.remove_internal(data)
251    }
252
253    fn pop_front_internal(&mut self) -> Option<NonNull<G::EntryType>> {
254        let head = self.head?;
255        // SAFETY: The head is on the list as we just got it from there and it cannot change.
256        unsafe { self.remove(head.as_ref()) };
257        Some(head)
258    }
259
260    /// Get and Remove the first element of the list.
261    pub fn pop_front(&mut self) -> Option<NonNull<G::EntryType>> {
262        self.pop_front_internal()
263    }
264
265    ///  Just Get and not remove the first element of the list.
266    pub(crate) fn front(&self) -> Option<NonNull<G::EntryType>> {
267        self.head
268    }
269
270    /// Just Get and not remove the last element of the list.
271    pub(crate) fn back(&self) -> Option<NonNull<G::EntryType>> {
272        // SAFETY: The links of head are owned by the list, so it is safe to get a reference.
273        unsafe { &*G::get_links(self.head?.as_ref()).entry.get() }.prev
274    }
275
276    /// Returns a cursor starting on the first element of the list.
277    pub(crate) fn cursor_front(&self) -> Cursor<'_, G> {
278        Cursor::new(self, self.front())
279    }
280
281    /// Returns a cursor starting on the last element of the list.
282    pub(crate) fn cursor_back(&self) -> Cursor<'_, G> {
283        Cursor::new(self, self.back())
284    }
285
286    /// Returns a mut cursor starting on the first element of the list.
287    pub fn cursor_front_mut(&mut self) -> CursorMut<'_, G> {
288        CursorMut::new(self, self.front())
289    }
290}

292struct CommonCursor<G: GetLinks> {
293    cur: Option<NonNull<G::EntryType>>,
294}
295
296impl<G: GetLinks> CommonCursor<G> {
297    const fn new(cur: Option<NonNull<G::EntryType>>) -> Self {
298        Self { cur }
299    }
300
301    fn move_next(&mut self, list: &RawList<G>) {
302        // `move_next` stops at None when reaching the end of list,
303        // because `raw_list::Iterator` stops at None.
304        // But move to next a further step beyond the ending None,
305        // the cursor starts from the head again.
306        match self.cur.take() {
307            None => self.cur = list.head,
308            Some(cur) => {
309                if let Some(head) = list.head {
310                    // SAFETY: We have a shared ref to the linked list, so the links can't change.
311                    let links = unsafe { &*G::get_links(cur.as_ref()).entry.get() };
312                    if !ptr::addr_eq(links.next.unwrap().as_ptr(), head.as_ptr()) {
313                        self.cur = links.next;
314                    }
315                }
316            }
317        }
318    }
319
320    fn move_prev(&mut self, list: &RawList<G>) {
321        match list.head {
322            None => self.cur = None,
323            Some(head) => {
324                let next = match self.cur.take() {
325                    None => head,
326                    Some(cur) => {
327                        if ptr::addr_eq(cur.as_ptr(), head.as_ptr()) {
328                            return;
329                        }
330                        cur
331                    }
332                };
333                // SAFETY: There's a shared ref to the list, so the links can't change.
334                let links = unsafe { &*G::get_links(next.as_ref()).entry.get() };
335                self.cur = links.prev;
336            }
337        }
338    }
339}

349/// A list cursor that allows traversing a linked list and inspecting elements.
350pub(crate) struct Cursor<'a, G: GetLinks> {
351    cursor: CommonCursor<G>,
352    list: &'a RawList<G>,
353}
354
355impl<'a, G: GetLinks> Cursor<'a, G> {
356    pub(crate) fn new(list: &'a RawList<G>, cur: Option<NonNull<G::EntryType>>) -> Self {
357        Self {
358            list,
359            cursor: CommonCursor::new(cur),
360        }
361    }
362
363    /// Returns the element the cursor is currently positioned on.
364    pub(crate) fn current(&self) -> Option<&'a G::EntryType> {
365        let cur = self.cursor.cur?;
366        // SAFETY: Objects must be kept alive while on the list.
367        Some(unsafe { &*cur.as_ptr() })
368    }
369
370    /// Returns the element pointer the cursor is currently positioned on.
371    pub(crate) fn current_ptr(&self) -> Option<NonNull<G::EntryType>> {
372        self.cursor.cur
373    }
374
375    /// Moves the cursor to the next element.
376    pub(crate) fn move_next(&mut self) {
377        self.cursor.move_next(self.list);
378    }
379
380    pub fn peek_next(&self) -> Option<&G::EntryType> {
381        let mut new = CommonCursor::new(self.cursor.cur);
382        new.move_next(self.list);
383        // SAFETY: Objects must be kept alive while on the list.
384        Some(unsafe { &*new.cur?.as_ptr() })
385    }
386
387    pub fn peek_prev(&self) -> Option<&G::EntryType> {
388        let mut new = CommonCursor::new(self.cursor.cur);
389        new.move_prev(self.list);
390        // SAFETY: Objects must be kept alive while on the list.
391        Some(unsafe { &*new.cur?.as_ptr() })
392    }
393
394    /// Moves the cursor to the prev element.
395    pub(crate) fn move_prev(&mut self) {
396        self.cursor.move_prev(self.list);
397    }
398}

400pub struct CursorMut<'a, G: GetLinks> {

402    pub(crate) list: &'a mut RawList<G>,
403}

405impl<'a, G: GetLinks> CursorMut<'a, G> {
406    fn new(list: &'a mut RawList<G>, cur: Option<NonNull<G::EntryType>>) -> Self {
407        Self {
408            list,
409            cursor: CommonCursor::new(cur),
410        }
411    }
412
413    pub unsafe fn current_mut(&mut self) -> Option<&mut G::EntryType> {
414        let cur = self.cursor.cur?;
415        // SAFETY: Objects must be kept alive while on the list.
416        Some(unsafe { &mut *cur.as_ptr() })
417    }
418
419    pub fn current(&self) -> Option<&G::EntryType> {
420        let cur = self.current_ptr()?;
421        // SAFETY: Objects must be kept alive while on the list.
422        Some(unsafe { &mut *cur.as_ptr() })
423    }
424
425    /// Returns the element pointer the cursor is currently positioned on.
426    pub(crate) fn current_ptr(&self) -> Option<NonNull<G::EntryType>> {
427        self.cursor.cur
428    }
429
430    /// Removes the entry the cursor is pointing to and advances the cursor to the next entry. It
431    /// returns a raw pointer to the removed element (if one is removed).
432    pub fn remove_current(&mut self) -> Option<NonNull<G::EntryType>> {
433        let entry = self.cursor.cur?;
434        self.cursor.move_next(self.list);
435        // SAFETY: The entry is on the list as we just got it from there and it cannot change.
436        unsafe { self.list.remove(entry.as_ref()) };
437        Some(entry)
438    }
439
440    /// Returns the element immediately after the one the cursor is positioned on.
441    ///
442    /// # Safety
443    ///
444    /// For `Box` or `Arc` that has unique access to the data behind, the method is safe.
445    /// For `Arc` whose strong count is not 1, the method is not safe because it
446    /// violates the safety requirements of [`Arc`].
447    pub unsafe fn peek_next(&mut self) -> Option<&mut G::EntryType> {
448        let mut new = CommonCursor::new(self.cursor.cur);
449        new.move_next(self.list);
450        // SAFETY: Objects must be kept alive while on the list.
451        Some(unsafe { &mut *new.cur?.as_ptr() })
452    }
453
454    /// Returns the element immediately before the one the cursor is positioned on.
455    ///
456    /// # Safety
457    ///
458    /// For `Box` or `Arc` that has unique access to the data behind, the method is safe.
459    /// For `Arc` whose strong count is not 1, the method is not safe because it
460    /// violates the safety requirements of [`Arc`].
461    pub unsafe fn peek_prev(&mut self) -> Option<&mut G::EntryType> {
462        let mut new = CommonCursor::new(self.cursor.cur);
463        new.move_prev(self.list);
464        // SAFETY: Objects must be kept alive while on the list.
465        Some(unsafe { &mut *new.cur?.as_ptr() })
466    }
467
468    pub fn move_next(&mut self) {
469        self.cursor.move_next(self.list);
470    }
471
472    pub fn move_prev(&mut self) {
473        self.cursor.move_prev(self.list);
474    }

485/// An iterator for the linked list.
486pub struct Iterator<'a, G: GetLinks> {
487    cursor_front: Cursor<'a, G>,
488    cursor_back: Cursor<'a, G>,
489}
490
491impl<'a, G: GetLinks> Iterator<'a, G> {
492    const fn new(cursor_front: Cursor<'a, G>, cursor_back: Cursor<'a, G>) -> Self {
493        Self {
494            cursor_front,
495            cursor_back,
496        }
497    }
498}
499
500impl<'a, G: GetLinks> iter::Iterator for Iterator<'a, G> {
501    type Item = &'a G::EntryType;
502
503    fn next(&mut self) -> Option<Self::Item> {
504        let ret = self.cursor_front.current()?;
505        self.cursor_front.move_next();
506        Some(ret)
507    }
508}
509
510impl<G: GetLinks> iter::DoubleEndedIterator for Iterator<'_, G> {
511    fn next_back(&mut self) -> Option<Self::Item> {
512        let ret = self.cursor_back.current()?;
513        self.cursor_back.move_prev();
514        Some(ret)
515    }
516}

518#[cfg(test)]
519mod tests {
520    extern crate alloc;
521    use alloc::{boxed::Box, vec::Vec};
522    use core::ptr::NonNull;
523
524    struct Example {
525        links: super::Links<Self>,
526    }
527
528    // SAFETY: This is the only adapter that uses `Example::links`.
529    impl super::GetLinks for Example {
530        type EntryType = Self;
531        fn get_links(obj: &Self) -> &super::Links<Self> {
532            &obj.links
533        }
534    }
535
536    fn build_vector(size: usize) -> Vec<Box<Example>> {
537        let mut v = Vec::new();
538        v.reserve(size);
539        for _ in 0..size {
540            v.push(Box::new(Example {
541                links: super::Links::new(),
542            }));
543        }
544        v
545    }
546
547    #[track_caller]
548    fn assert_list_contents(v: &[Box<Example>], list: &super::RawList<Example>) {
549        let n = v.len();
550
551        // Assert that the list is ok going forward.
552        let mut count = 0;
553        for (i, e) in list.iter().enumerate() {
554            assert!(core::ptr::eq(e, &*v[i]));
555            count += 1;
556        }
557        assert_eq!(count, n);
558
559        // Assert that the list is ok going backwards.
560        let mut count = 0;
561        for (i, e) in list.iter().rev().enumerate() {
562            assert!(core::ptr::eq(e, &*v[n - 1 - i]));
563            count += 1;
564        }
565        assert_eq!(count, n);
566    }
567
568    #[track_caller]
569    fn test_each_element(
570        min_len: usize,
571        max_len: usize,
572        test: impl Fn(&mut Vec<Box<Example>>, &mut super::RawList<Example>, usize, Box<Example>),
573    ) {
574        for n in min_len..=max_len {
575            for i in 0..n {
576                let extra = Box::new(Example {
577                    links: super::Links::new(),
578                });
579                let mut v = build_vector(n);
580                let mut list = super::RawList::<Example>::new();
581
582                // Build list.
583                for j in 0..n {
584                    // SAFETY: The entry was allocated above, it's not in any lists yet, is never
585                    // moved, and outlives the list.
586                    unsafe { list.push_back(NonNull::from(&*v[j])) };
587                }
588
589                // Call the test case.
590                test(&mut v, &mut list, i, extra);
591
592                // Check that the list is ok.
593                assert_list_contents(&v, &list);
594            }
595        }
596    }

598    #[test]
599    fn test_push_back() {
600        const MAX: usize = 10;
601        let v = build_vector(MAX);
602        let mut list = super::RawList::<Example>::new();
603
604        for n in 1..=MAX {
605            // SAFETY: The entry was allocated above, it's not in any lists yet, is never moved,
606            // and outlives the list.
607            unsafe { list.push_back(NonNull::from(&*v[n - 1])) };
608            assert_list_contents(&v[..n], &list);
609        }

616        let mut list = super::RawList::<Example>::new();
617
618        for n in 1..=MAX {
619            // SAFETY: The entry was allocated above, it's not in any lists yet, is never moved,
620            // and outlives the list.
621            println!("push front: {}", MAX - n);
622            unsafe { list.push_front(NonNull::from(&*v[MAX - n])) };
623            assert_list_contents(&v[MAX - n..MAX], &list);
